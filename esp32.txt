// ==================== Status LED Helpers ====================
void ledOn() {
  if (STATUS_LED_PIN >= 0) digitalWrite(STATUS_LED_PIN, HIGH);
}

void ledOff() {
  if (STATUS_LED_PIN >= 0) digitalWrite(STATUS_LED_PIN, LOW);
}

void ledBlink(int times, int delayMs = 200) {
  for (int i = 0; i < times; i++) {
    ledOn();
    delay(delayMs);
    ledOff();
    delay(delayMs);
  }
}

// ==================== Initialize Camera ====================
bool initCamera() {
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;  // 20MHz (stable for OV3660)
  config.pixel_format = PIXFORMAT_JPEG;
  config.frame_size = FRAME_SIZE;
  config.jpeg_quality = JPEG_QUALITY;
  config.fb_count = FB_COUNT;
  config.grab_mode = CAMERA_GRAB_WHEN_EMPTY;

  // Initialize camera
  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("Camera init failed: 0x%x\n", err);
    return false;
  }

  // Configure sensor for optimal quality
  sensor_t* s = esp_camera_sensor_get();
  
  // Basic image tuning
  s->set_brightness(s, 0);      // -2 to 2
  s->set_contrast(s, 0);        // -2 to 2
  s->set_saturation(s, 1);      // Slight saturation boost helps color detection
  s->set_special_effect(s, 0);  // No effect
  
  // Auto controls ON (important for outdoor use)
  s->set_whitebal(s, 1);        // Auto white balance
  s->set_awb_gain(s, 1);        // AWB gain
  s->set_exposure_ctrl(s, 1);   // Auto exposure
  s->set_gain_ctrl(s, 1);       // Auto gain
  s->set_aec2(s, 1);            // Advanced AEC
  
  // Quality enhancements
  s->set_raw_gma(s, 1);         // Gamma correction
  s->set_lenc(s, 1);            // Lens correction
  s->set_dcw(s, 1);             // Downsize enable
  
  // Exposure bias: slightly brighter for plate recognition (0-1200)
  s->set_aec_value(s, 400);
  
  // Sharpness: helps OCR read plates
  s->set_sharpness(s, 1);       // 0 = auto, 1 = slight sharpen
  
  // Mirror/flip based on your mount orientation
  s->set_hmirror(s, 0);         // 1 = horizontal mirror
  s->set_vflip(s, 0);           // 1 = vertical flip

  Serial.println("Camera initialized: OV3660 SVGA 800x600");
  return true;
}

// ==================== MJPEG Stream Handler ====================
void handleStream() {
  Serial.println("Stream client connected");
  
  WiFiClient client = server.client();
  
  // Build HTTP response header
  String head = "HTTP/1.1 200 OK\r\n";
  head += "Content-Type: multipart/x-mixed-replace; boundary=--frameboundary\r\n";
  head += "Cache-Control: no-cache, no-store, must-revalidate\r\n";
  head += "Pragma: no-cache\r\n";
  head += "Expires: 0\r\n";
  head += "Connection: close\r\n";
  head += "Access-Control-Allow-Origin: *\r\n";
  head += "\r\n";
  
  client.print(head);
  
  unsigned long last_frame_time = 0;
  unsigned long stream_start = millis();
  int frame_delay_ms = (MAX_FPS > 0) ? (1000 / MAX_FPS) : 0;
  int frames_sent = 0;
  int consecutive_failures = 0;
  const int MAX_CONSECUTIVE_FAILURES = 10;
  
  stream_active = true;
  
  while (client.connected() && stream_active) {
    // Feed watchdog
    esp_task_wdt_reset();
    
    // Frame rate limiting
    unsigned long now = millis();
    if (frame_delay_ms > 0 && (now - last_frame_time < frame_delay_ms)) {
      delay(1);
      continue;
    }
    last_frame_time = now;
    
    // Capture frame
    camera_fb_t* fb = esp_camera_fb_get();
    if (!fb) {
      consecutive_failures++;
      if (consecutive_failures > MAX_CONSECUTIVE_FAILURES) {
        Serial.println("Too many capture failures, stopping stream");
        break;
      }
      delay(10);
      continue;
    }
    consecutive_failures = 0;
    
    // Validate JPEG data: check for valid JPEG header
    bool valid_jpeg = (fb->len > 100) && 
                      (fb->buf[0] == 0xFF) && 
                      (fb->buf[1] == 0xD8);
    
    if (!valid_jpeg) {
      Serial.printf("Invalid JPEG header: %02X %02X (len=%d)\n", 
                    fb->buf[0], fb->buf[1], fb->len);
      esp_camera_fb_return(fb);
      delay(5);
      continue;
    }
    
    // Send MJPEG frame boundary
    client.print("--frameboundary\r\n");
    client.print("Content-Type: image/jpeg\r\n");
    client.printf("Content-Length: %d\r\n", fb->len);
    client.print("\r\n");
    
    // Send image data
    size_t written = client.write(fb->buf, fb->len);
    client.print("\r\n");
    
    if (written != fb->len) {
      Serial.printf("Write incomplete: %d/%d bytes\n", written, fb->len);
      esp_camera_fb_return(fb);
      break;
    }
    
    frames_sent++;
    
    // Print stats every 30 frames
    if (frames_sent % 30 == 0) {
      float elapsed = (millis() - stream_start) / 1000.0;
      if (elapsed > 0) {
        current_fps = frames_sent / elapsed;
        Serial.printf("FPS: %.1f | Frame: %d bytes | Total: %d frames | Free heap: %d\n", 
                      current_fps, fb->len, frames_sent, ESP.getFreeHeap());
      }
    }
    
    // Return frame buffer
    esp_camera_fb_return(fb);
    
    // Yield to allow WiFi and other tasks to run
    yield();
    
    // Check if client disconnected
    if (!client.connected()) {
      Serial.println("Stream client disconnected");
      break;
    }
  }
  
  stream_active = false;
  
  if (frames_sent > 0) {
    float total_time = (millis() - stream_start) / 1000.0;
    Serial.printf("Stream ended. Sent %d frames in %.1f seconds (%.1f avg FPS)\n", 
                  frames_sent, total_time, frames_sent / total_time);
  }
}

// ==================== Single Frame Capture ====================
void handleCapture() {
  camera_fb_t* fb = esp_camera_fb_get();
  if (!fb) {
    server.send(500, "text/plain", "Capture failed");
    return;
  }
  
  // Validate JPEG
  if (fb->len < 100 || fb->buf[0] != 0xFF || fb->buf[1] != 0xD8) {
    esp_camera_fb_return(fb);
    server.send(500, "text/plain", "Invalid JPEG data");
    return;
  }
  
  WiFiClient client = server.client();
  
  String head = "HTTP/1.1 200 OK\r\n";
  head += "Content-Type: image/jpeg\r\n";
  head += "Content-Length: " + String(fb->len) + "\r\n";
  head += "Cache-Control: no-cache\r\n";
  head += "\r\n";
  client.print(head);
  client.write(fb->buf, fb->len);
  
  esp_camera_fb_return(fb);
}

// ==================== Status/Info Endpoint ====================
void handleStatus() {
  String json = "{";
  json += "\"camera\":\"OV3660\",";
  json += "\"resolution\":\"800x600\",";
  json += "\"fps\":" + String(current_fps, 1) + ",";
  json += "\"stream_active\":" + String(stream_active ? "true" : "false") + ",";
  json += "\"free_heap\":" + String(ESP.getFreeHeap()) + ",";
  json += "\"psram_size\":" + String(ESP.getPsramSize()) + ",";
  json += "\"free_psram\":" + String(ESP.getFreePsram()) + ",";
  json += "\"wifi_rssi\":" + String(WiFi.RSSI()) + ",";
  json += "\"uptime\":" + String(millis() / 1000);
  json += "}";
  
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.send(200, "application/json", json);
}

// ==================== Restart Endpoint ====================
void handleRestart() {
  server.send(200, "text/plain", "Restarting...");
  delay(100);
  ESP.restart();
}

// ==================== Connect to WiFi ====================
bool connectWiFi() {
  Serial.print("Connecting to WiFi: ");
  Serial.print(ssid);
  
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 40) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n[OK] WiFi connected!");
    Serial.print("     IP Address: ");
    Serial.println(WiFi.localIP());
    Serial.print("     Gateway:    ");
    Serial.println(WiFi.gatewayIP());
    Serial.print("     RSSI:       ");
    Serial.print(WiFi.RSSI());
    Serial.println(" dBm");
    
    // Set WiFi auto-reconnect
    WiFi.setAutoReconnect(true);
    WiFi.persistent(true);
    
    // Disable power saving for lower latency
    esp_wifi_set_ps(WIFI_PS_NONE);
    
    wifi_was_connected = true;
    return true;
  } else {
    Serial.println("\n[FAIL] WiFi connection failed");
    wifi_was_connected = false;
    return false;
  }
}

// ==================== WiFi Keep-Alive Check ====================
void checkWiFi() {
  // Check if disconnected
  if (WiFi.status() != WL_CONNECTED) {
    if (wifi_was_connected) {
      Serial.println("[WARN] WiFi disconnected! Attempting reconnect...");
      wifi_was_connected = false;
      ledBlink(3, 100);  // 3 fast blinks = connection lost
    }
    
    // Attempt reconnection periodically
    unsigned long now = millis();
    if (now - last_reconnect_attempt > WIFI_RECONNECT_INTERVAL) {
      last_reconnect_attempt = now;
      Serial.println("[INFO] Attempting WiFi reconnect...");
      
      if (connectWiFi()) {
        Serial.println("[OK] WiFi reconnected!");
        ledBlink(2, 300);  // 2 slow blinks = reconnected
      }
    }
  } else if (!wifi_was_connected) {
    wifi_was_connected = true;
    Serial.println("[OK] WiFi connection restored");
    Serial.print("     IP: ");
    Serial.println(WiFi.localIP());
  }
}

// ==================== Setup ====================
void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n");
  Serial.println("========================================");
  Serial.println("  XIAO ESP32S3 OV3660 Road Monitor");
  Serial.println("========================================");
  
  // Initialize status LED
  if (STATUS_LED_PIN >= 0) {
    pinMode(STATUS_LED_PIN, OUTPUT);
    ledOff();
  }
  
  // Configure watchdog
  esp_task_wdt_deinit();                      // Clear any existing TWDT from bootloader
  esp_task_wdt_config_t wdt_config = {
    .timeout_ms = WDT_TIMEOUT * 1000,
    .idle_core_mask = 0,
    .trigger_panic = true
  };
  esp_task_wdt_init(&wdt_config);
  esp_task_wdt_add(NULL);
  
  // Initialize camera
  Serial.print("[INIT] Camera... ");
  if (!initCamera()) {
    Serial.println("FAILED!");
    Serial.println("Check camera connections and restart.");
    ledBlink(5, 150);  // 5 fast blinks = camera error
    while (1) {
      delay(1000);
    }
  }
  Serial.println("OK");
  
  // Connect to WiFi
  if (!connectWiFi()) {
    Serial.println("[INFO] Starting AP mode as fallback...");
    WiFi.softAP("XIAO_Camera", "12345678");
    Serial.print("       AP IP: ");
    Serial.println(WiFi.softAPIP());
    ledBlink(1, 500);  // 1 long blink = AP mode
  } else {
    ledBlink(2, 200);  // 2 blinks = All OK
  }
  
  // Setup web server routes
  server.on("/", handleRoot);
  server.on("/stream", handleStream);
  server.on("/capture", handleCapture);
  server.on("/status", handleStatus);
  server.on("/restart", handleRestart);
  server.onNotFound(handleNotFound);
  
  server.begin();
  
  Serial.println("----------------------------------------");
  Serial.println("[READY] Server started");
  Serial.println("----------------------------------------");
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.print("  Web UI:     http://");
    Serial.println(WiFi.localIP());
    Serial.print("  Stream URL: http://");
    Serial.print(WiFi.localIP());
    Serial.println("/stream");
    Serial.print("  Status:     http://");
    Serial.print(WiFi.localIP());
    Serial.println("/status");
  } else {
    Serial.print("  Web UI:     http://");
    Serial.println(WiFi.softAPIP());
    Serial.print("  Stream URL: http://");
    Serial.print(WiFi.softAPIP());
    Serial.println("/stream");
  }
  Serial.println("----------------------------------------");
  Serial.println("Waiting for Python server connection...");
}

// ==================== Main Loop ====================
void loop() {
  // Feed watchdog
  esp_task_wdt_reset();
  
  // Handle web server clients
  server.handleClient();
  
  // Check WiFi connection health
  checkWiFi();
  
  // Small delay to prevent tight-looping
  delay(5);
}
